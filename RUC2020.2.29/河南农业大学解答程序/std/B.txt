#include<bits/stdc++.h>
using namespace std ;
typedef long long ll;
const int maxn=25+10;
const int mod=1e9+7;
int d[(1<<20)+ 5][25];
int x[maxn],y[maxn];
map<string,int> mp;
int sz[maxn];//оо└Я
int cal(int a,int b){
    return abs(x[a]-x[b])+abs(y[a]-y[b]);
}
int main(){
    int n;
    scanf("%d",&n);
    string a;
    int cnt=-1;
    for(int i=1;i<=n;i++){
        cin>>x[i]>>y[i]>>a;
        if(!mp.count(a))mp[a]=++cnt;
        sz[i]=mp[a];
    }
    memset(d,0x3f,sizeof(d));
    for(int i=1;i<=n;i++)d[1<<sz[i]][i]=cal(i,0);
    for(int s=0;s < (1<<cnt+1);s++){
       for(int i=1;i<=n;i++){
            if(s & (1 << sz[i])){
               for(int j=1;j<=n;j++){
                   if(s & (1<<sz[j]))
                         d[s][i]=min(d[s][i],d[s^(1<<sz[i])][j]+cal(i,j));
                }
           }
        }
   }
   int ans=0x3f3f3f3f;
   for(int i=1;i<=n;i++)ans=min(ans,d[(1<<cnt+1)-1][i] + cal(i,0));
   cout<<ans;
   return 0;
}
